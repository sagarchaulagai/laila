# 3qn. Write a MATLAB program that calculates and plot the poles and zeros of the accumulator
circuit whose difference equation is

y[n] = x[n]+ y[n -1].
% CRN: 021-389 - Pole-Zero Plot for Accumulator

b = 1;
a = [1 -1];

[z, p, k] = tf2zp(b, a);

zplane(z, p);
title('CRN: 021-389 - Accumulator Pole-Zero Plot');

disp('Zeros:'); disp(z);
disp('Poles:'); disp(p);
disp('Gain:'); disp(k);
